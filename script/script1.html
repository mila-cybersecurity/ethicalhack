Detección de Inicio de Sesión Sospechoso

Este sistema verifica si un usuario ha realizado tres o más intentos fallidos de inicio de sesión. 


# Código de detección de inicio de sesión
with open("usuarios_inicio_sesion.txt", "r", encoding="utf8") as archivo:
    archivo_texto = archivo.read()
nombres_usuarios = archivo_texto.split()

def verificar_inicio_sesion(lista_inicios, usuario_actual):
    if usuario_actual in lista_inicios:
        contador = 0
        for i in lista_inicios:
            if i == usuario_actual:
                contador += 1
        if contador >= 3:
            return "Haz hecho 3 o más intentos de inicio de sesión. Tu cuenta ha sido bloqueada"
        else:
            return "Estás logueado"
    else:
        return "Usuario inexistente"

# Verificar inicio de sesión de un usuario específico
print(verificar_inicio_sesion(nombres_usuarios, "López"))
